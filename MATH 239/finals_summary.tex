\subsection*{Connectedness}
\begineasylist

# Fix vertex $v$ in $G$; 
## $\forall$ vertex $w$ in $G$, $\exists$ path between $v,w$ $\implies$ $G$ is \textbf{connected}

# Let $X \subset V(G)$;
## ``Cut induced by $X$'' = set of edges with exactly one vertex $\in X$
## \textbf{Theorem:} $G$ is \underline{not connected} $\iff$ $\exists$ $X$ such that cut induced by $X$ is empty

# ``Eulerian circuit'' = closed walk that contains every edge exactly once

# \textbf{Theorem:} $G$ has Eulerian circuit $\iff$ $G$ is connected AND every vertex has even degree

# \textbf{Lemma:} $G$ is connected AND $e$ is a bridge $\implies$ $G - e$ has exactly 2 components

# \textbf{Theorem:} $e$ is a bridge $\iff$ $e$ is not contained in any cycle

## \textbf{Corollary:} $\exists$ 2 distinct paths between $u,v$ in $G$ $\implies$ $G$ contains a cycle

\subsection*{Planarity}
\begineasylist

# \textbf{Handshaking Lemma for Faces (Faceshaking Lemma):}
\[ \sum_{f \in faces} deg(f) = 2|E(G)| \]

# \textbf{Euler's Formula:} let $G$ be a connected graph with $p$ vertices and $q$ edges; if G has a planar embedding with $f$ faces, then
\[ p - q + f = 2 \]

# \textbf{Theorem:} a graph is planar $\iff$ it can be drawn on the surface of a sphere

# ``Platonic graph'' = graph whose planar embedding has vertices all with degree $d \geq 3$ and faces all with degree $d^* \geq 3$

# \textbf{Theorem:} there are exactly 5 platonic graphs

# \textbf{Lemma:} $(d, d^*)$ pairs are: $(3,3), (3,4), (4,3), (3,5), (5,3)$

# \textbf{Lemma:} $G$ is a platonic graph with $p$ vertices of degree $d$, $q$ edges, and $f$ faces of degree $d^*$, then:
\begin{align*}
q = \frac{2dd^*}{2d + 2d^* - dd*} \qquad \qquad \qquad
p = \frac{2q}{d} \qquad \qquad \qquad
f = \frac{2q}{d^*}
\end{align*}
    
# \textbf{Theorem:} if a graph is connected and planar with $p \geq 3$ vertices and $q$ edges, then $q \leq 3p -6$
## \textbf{Corollary:} a planar graph has a vertex of degree $< 6$

# Note: $K_5$ and $K_{3,3}$ are not planar

# \textbf{Kuratowski's Theorem:} $G$ is \underline{not planar} $\iff$ $G$ contains a edge subdivision of $K_5$ or $K_{3,3}$

# \textbf{Theorem:} $G$ is 2-colourable $\iff$ $G$ is bipartite

# \textbf{Theorem:} $K_n$ is $n$-colourable

# \textbf{Theorem:} every vertex of $G$ has degree $\leq d$ $\implies$ $G$ is $(d+1)$-colourable

# \textbf{Four Colour Theorem:} every planar graph is 4-colourable
